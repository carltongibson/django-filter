125class IsoDateTimeField(forms.DateTimeField): 

126 """ 

127 Supports 'iso-8601' date format too which is out the scope of 

128 the ``datetime.strptime`` standard library 

129 

130 # ISO 8601: ``http://www.w3.org/TR/NOTE-datetime`` 

131 

132 Based on Gist example by David Medina https://gist.github.com/copitux/5773821 

133 """ 

134 

135 ISO_8601 = "iso-8601" 

136 input_formats = [ISO_8601] 

137 

138 def strptime(self, value, format): 

139 value = force_str(value) 

140 

141 if format == self.ISO_8601: 

142 parsed = parse_datetime(value) 

143 if parsed is None: # Continue with other formats if doesn't match 

144 raise ValueError 

145 return handle_timezone(parsed) 

146 return super().strptime(value, format)

149class BaseCSVField(forms.Field): 

150 """ 

151 Base field for validating CSV types. Value validation is performed by 

152 secondary base classes. 

153 

154 ex:: 

155 class IntegerCSVField(BaseCSVField, filters.IntegerField): 

156 pass 

157 

158 """ 

159 

160 base_widget_class = BaseCSVWidget 

161 

162 def __init__(self, *args, **kwargs): 

163 widget = kwargs.get("widget") or self.widget 

164 kwargs["widget"] = self._get_widget_class(widget) 

165 

166 super().__init__(*args, **kwargs) 

167 

168 def _get_widget_class(self, widget): 

169 # passthrough, allows for override 

170 if isinstance(widget, BaseCSVWidget) or ( 

171 isinstance(widget, type) and issubclass(widget, BaseCSVWidget) 

172 ): 

173 return widget 

174 

175 # complain since we are unable to reconstruct widget instances 

176 assert isinstance( 

177 widget, type 

178 ), "'%s.widget' must be a widget class, not %s." % ( 

179 self.__class__.__name__, 

180 repr(widget), 

181 ) 

182 

183 bases = ( 

184 self.base_widget_class, 

185 widget, 

186 ) 

187 return type(str("CSV%s" % widget.__name__), bases, {}) 

188 

189 def clean(self, value): 

190 if value in self.empty_values and self.required: 

191 raise forms.ValidationError( 

192 self.error_messages["required"], code="required" 

193 ) 

194 

195 if value is None: 

196 return None 

197 return [super(BaseCSVField, self).clean(v) for v in value]

220class ChoiceIterator(BaseChoiceIterator if DJANGO_50 else object): 

221 # Emulates the behavior of ModelChoiceIterator, but instead wraps 

222 # the field's _choices iterable. 

223 

224 def __init__(self, field, choices): 

225 self.field = field 

226 self.choices = choices 

227 

228 def __iter__(self): 

229 if self.field.empty_label is not None: 

230 yield ("", self.field.empty_label) 

231 if self.field.null_label is not None: 

232 yield (self.field.null_value, self.field.null_label) 

233 if DJANGO_50: 

234 yield from normalize_choices(self.choices) 

235 else: 

236 yield from self.choices 

237 

238 def __len__(self): 

239 add = 1 if self.field.empty_label is not None else 0 

240 add += 1 if self.field.null_label is not None else 0 

241 return len(self.choices) + add

244class ModelChoiceIterator(forms.models.ModelChoiceIterator): 

245 # Extends the base ModelChoiceIterator to add in 'null' choice handling. 

246 # This is a bit verbose since we have to insert the null choice after the 

247 # empty choice, but before the remainder of the choices. 

248 

249 def __iter__(self): 

250 iterable = super().__iter__() 

251 

252 if self.field.empty_label is not None: 

253 yield next(iterable) 

254 if self.field.null_label is not None: 

255 yield (self.field.null_value, self.field.null_label) 

256 yield from iterable 

257 

258 def __len__(self): 

259 add = 1 if self.field.null_label is not None else 0 

260 return super().__len__() + add
