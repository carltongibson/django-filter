33def remote_queryset(field): 

34 """ 

35 Get the queryset for the other side of a relationship. This works 

36 for both `RelatedField`s and `ForeignObjectRel`s. 

37 """ 

38 model = field.related_model 

39

42 return model._default_manager.all()

45 return model._default_manager.complex_filter(limit_choices_to)

76class FilterSetMetaclass(type): 

77 def __new__(cls, name, bases, attrs): 

78 attrs["declared_filters"] = cls.get_declared_filters(bases, attrs) 

79 

80 new_class = super().__new__(cls, name, bases, attrs) 

81 new_class._meta = FilterSetOptions(getattr(new_class, "Meta", None)) 

82 new_class.base_filters = new_class.get_filters() 

83 

84 return new_class

86 @classmethod 

87 def get_declared_filters(cls, bases, attrs): 

88 filters = [ 

89 (filter_name, attrs.pop(filter_name)) 

90 for filter_name, obj in list(attrs.items()) 

91 if isinstance(obj, Filter) 

92 ] 

93 

94 # Default the `filter.field_name` to the attribute name on the filterset 

95 for filter_name, f in filters: 

96 if getattr(f, "field_name", None) is None: 

97 f.field_name = filter_name 

98 

99 filters.sort(key=lambda x: x[1].creation_counter) 

100 

101 # Ensures a base class field doesn't override cls attrs, and maintains 

102 # field precedence when inheriting multiple parents. e.g. if there is a 

103 # class C(A, B), and A and B both define 'field', use 'field' from A. 

104 known = set(attrs) 

105 

106 def visit(name): 

107 known.add(name) 

108 return name 

109 

110 base_filters = [ 

111 (visit(name), f) 

112 for base in bases 

113 if hasattr(base, "declared_filters") 

114 for name, f in base.declared_filters.items() 

115 if name not in known 

116 ] 

117 

118 return OrderedDict(base_filters + filters)

224 def filter_queryset(self, queryset): 

225 """ 

226 Filter the queryset with the underlying form's `cleaned_data`. You must 

227 call `is_valid()` or `errors` before calling this method. 

228 

229 This method should be overridden if additional filtering needs to be 

230 applied to the queryset before it is cached. 

231 """ 

232 for name, value in self.form.cleaned_data.items(): 

233 queryset = self.filters[name].filter(queryset, value) 

234 assert isinstance( 

235 queryset, models.QuerySet 

236 ), "Expected '%s.%s' to return a QuerySet, but got a %s instead." % ( 

237 type(self).__name__, 

238 name, 

239 type(queryset).__name__, 

240 ) 

241 return queryset

277 @classmethod 

278 def get_fields(cls): 

279 """ 

280 Resolve the 'fields' argument that should be used for generating filters on the 

281 filterset. This is 'Meta.fields' sans the fields in 'Meta.exclude'. 

282 """ 

283 model = cls._meta.model 

284 fields = cls._meta.fields 

285 exclude = cls._meta.exclude 

286 

287 assert not (fields is None and exclude is None), ( 

288 "Setting 'Meta.model' without either 'Meta.fields' or 'Meta.exclude' " 

289 "has been deprecated since 0.15.0 and is now disallowed. Add an explicit " 

290 "'Meta.fields' or 'Meta.exclude' to the %s class." % cls.__name__ 

291 ) 

292 

293 # Setting exclude with no fields implies all other fields. 

294 if exclude is not None and fields is None: 

295 fields = ALL_FIELDS 

296 

297 # Resolve ALL_FIELDS into all fields for the filterset's model. 

298 if fields == ALL_FIELDS: 

299 fields = get_all_model_fields(model) 

300 

301 # Remove excluded fields 

302 exclude = exclude or [] 

303 if not isinstance(fields, dict): 

304 fields = [ 

305 (f, [settings.DEFAULT_LOOKUP_EXPR]) for f in fields if f not in exclude 

306 ] 

307 else: 

308 fields = [(f, lookups) for f, lookups in fields.items() if f not in exclude] 

309 

310 return OrderedDict(fields)

312 @classmethod 

313 def get_filter_name(cls, field_name, lookup_expr): 

314 """ 

315 Combine a field name and lookup expression into a usable filter name. 

316 Exact lookups are the implicit default, so "exact" is stripped from the 

317 end of the filter name. 

318 """ 

319 filter_name = LOOKUP_SEP.join([field_name, lookup_expr]) 

320 

321 # This also works with transformed exact lookups, such as 'date__exact' 

322 _default_expr = LOOKUP_SEP + settings.DEFAULT_LOOKUP_EXPR 

323 if filter_name.endswith(_default_expr): 

324 filter_name = filter_name[: -len(_default_expr)] 

325 

326 return filter_name

328 @classmethod 

329 def get_filters(cls): 

330 """ 

331 Get all filters for the filterset. This is the combination of declared and 

332 generated filters. 

333 """ 

334 

335 # No model specified - skip filter generation 

336 if not cls._meta.model: 

337 return cls.declared_filters.copy() 

338 

339 # Determine the filters that should be included on the filterset. 

340 filters = OrderedDict() 

341 fields = cls.get_fields() 

342 undefined = [] 

343 

344 for field_name, lookups in fields.items(): 

345 field = get_model_field(cls._meta.model, field_name) 

346 

347 # warn if the field doesn't exist. 

348 if field is None: 

349 undefined.append(field_name) 

350 

351 for lookup_expr in lookups: 

352 filter_name = cls.get_filter_name(field_name, lookup_expr) 

353 

354 # If the filter is explicitly declared on the class, skip generation 

355 if filter_name in cls.declared_filters: 

356 filters[filter_name] = cls.declared_filters[filter_name] 

357 continue 

358 

359 if field is not None: 

360 filter_instance = cls.filter_for_field( 

361 field, field_name, lookup_expr 

362 ) 

363 if filter_instance is not None: 

364 filters[filter_name] = filter_instance 

365 

366 # Allow Meta.fields to contain declared filters *only* when a list/tuple 

367 if isinstance(cls._meta.fields, (list, tuple)): 

368 undefined = [f for f in undefined if f not in cls.declared_filters] 

369 

370 if undefined: 

371 raise TypeError( 

372 "'Meta.fields' must not contain non-model field names: %s" 

373 % ", ".join(undefined) 

374 ) 

375 

376 # Add in declared filters. This is necessary since we don't enforce adding 

377 # declared filters to the 'Meta.fields' option 

378 filters.update(cls.declared_filters) 

379 return filters

381 @classmethod 

382 def handle_unrecognized_field(cls, field_name, message): 

383 behavior = cls._meta.unknown_field_behavior 

384 if behavior == UnknownFieldBehavior.RAISE: 

385 raise AssertionError(message) 

386 elif behavior == UnknownFieldBehavior.WARN: 

387 warnings.warn( 

388 f"Unrecognized field type for '{field_name}'. Field will be ignored." 

389 ) 

390 elif behavior == UnknownFieldBehavior.IGNORE: 

391 pass 

392 else: 

393 raise ValueError(f"Invalid unknown_field_behavior: {behavior}")

395 @classmethod 

396 def filter_for_field(cls, field, field_name, lookup_expr=None): 

397 if lookup_expr is None: 

398 lookup_expr = settings.DEFAULT_LOOKUP_EXPR 

399 field, lookup_type = resolve_field(field, lookup_expr) 

400 

401 default = { 

402 "field_name": field_name, 

403 "lookup_expr": lookup_expr, 

404 } 

405 

406 filter_class, params = cls.filter_for_lookup(field, lookup_type) 

407 default.update(params) 

408 

409 if filter_class is None: 

410 cls.handle_unrecognized_field(field_name, ( 

411 "%s resolved field '%s' with '%s' lookup to an unrecognized field " 

412 "type %s. Try adding an override to 'Meta.filter_overrides'. See: " 

413 "https://django-filter.readthedocs.io/en/main/ref/filterset.html" 

414 "#customise-filter-generation-with-filter-overrides" 

415 ) % (cls.__name__, field_name, lookup_expr, field.__class__.__name__)) 

416 return None 

417 

418 return filter_class(**default)

420 @classmethod 

421 def filter_for_lookup(cls, field, lookup_type): 

422 DEFAULTS = dict(cls.FILTER_DEFAULTS) 

423 if hasattr(cls, "_meta"): 

424 DEFAULTS.update(cls._meta.filter_overrides) 

425 

426 data = try_dbfield(DEFAULTS.get, field.__class__) or {} 

427 filter_class = data.get("filter_class") 

428 params = data.get("extra", lambda field: {})(field) 

429 

430 # if there is no filter class, exit early 

431 if not filter_class: 

432 return None, {} 

433 

434 # perform lookup specific checks 

435 if lookup_type == "exact" and getattr(field, "choices", None): 

436 return ChoiceFilter, {"choices": field.choices} 

437 

438 if lookup_type == "isnull": 

439 data = try_dbfield(DEFAULTS.get, models.BooleanField) 

440 

441 filter_class = data.get("filter_class") 

442 params = data.get("extra", lambda field: {})(field) 

443 return filter_class, params

469 @classmethod 

470 def _csv_filter_class_name(cls, filter_class, lookup_type): 

471 """ 

472 Generate a suitable class name for a concrete filter class. This is not 

473 completely reliable, as not all filter class names are of the format 

474 <Type>Filter. 

475 

476 ex:: 

477 

478 FilterSet._csv_filter_class_name(DateTimeFilter, 'in') 

479 

480 returns 'DateTimeInFilter' 

481 

482 """ 

483 # DateTimeFilter => DateTime 

484 type_name = filter_class.__name__ 

485 if type_name.endswith("Filter"): 

486 type_name = type_name[:-6] 

487 

488 # in => In 

489 lookup_name = lookup_type.capitalize() 

490 

491 # DateTimeInFilter 

492 return str("%s%sFilter" % (type_name, lookup_name))

499def filterset_factory(model, filterset=FilterSet, fields=None): 

500 attrs = {"model": model} 

501 if fields is None: 

502 if getattr(getattr(filterset, "Meta", {}), "fields", None) is None: 

503 attrs["fields"] = ALL_FIELDS 

504 else: 

505 attrs["fields"] = fields 

506 bases = (filterset.Meta,) if hasattr(filterset, "Meta") else () 

507 Meta = type("Meta", bases, attrs) 

508 return type(filterset)( 

509 str("%sFilterSet" % model._meta.object_name), (filterset,), {"Meta": Meta}
